            for(int i=0; i<input.Length; i++)
            {
                if(!Char.IsLetter(input[i]))
                {
                    valid = false;
                    break;
                }
            }
            if(valid)
            {
                for(int j=0; j<input.Length; j++)
                {
                    if(j==0||j%2==0)
                    {
                        int ascii = input[j];
                        if(ascii==90)
                        {
                            ascii=65;
                        }
                        else if(ascii==122){
                            ascii=97;
                        }
                        else{
                            ascii++;
                        }
                        char replace = (char)ascii;
                            StringBuilder sb = new StringBuilder(en);
                            sb[j] = replace;
                            en = sb.ToString();
                    }
                }
                Console.WriteLine(input+en);
            }
            else
            {
                Console.WriteLine("Invalid input");

        public string InsuranceName{ set{insuranceName=value;} get{return insuranceName ;} }
        public double AmountCovered{ set{amountCovered=value;} get{return amountCovered ;} }
    }
    
    public class MotorInsurance : Insurance
    {
        private double idv;
        private float depPercent;
        
        public double Idv{ set{idv=value;} get{return idv ;} }
        public float DepPercent{ set{depPercent=value;} get{return depPercent ;} }
        
        public double calculatePremium()
        {
            double dPercent=Convert.ToDouble(DepPercent);
            Idv = AmountCovered - ((AmountCovered * dPercent) / 100);
            double result=Idv*0.03;
            return result;
        }
    }

    {
        private int policyTerm;
        private float benefitPercent;
        
        public int PolicyTerm{ set{policyTerm=value;} get{return policyTerm ;} }
        public float BenefitPercent{ set{benefitPercent=value;} get{return benefitPercent ;} }
        
        public double calculatePremium()
        {
            double result=(AmountCovered - ((AmountCovered* BenefitPercent)/100)) / PolicyTerm;
            return result;
        }

       {   get;  set;   }
        public double TotalPrice
        {   get;  set;   }

    }
    public class Billing: TextileShop
    {
        public double GenerateBill()
        {
            TotalPrice = 0.0;
            if(ClothesType == "Women")
            {	 	  	    	 	      	  	 	
                TotalPrice = Quantity * Price + 250;
            }
            else if (ClothesType == "Children")
            {
                TotalPrice = Quantity * Price + 50;
            }
            else if (ClothesType == "Men")
            {
                TotalPrice = Quantity * Price + 150;
            }
            return TotalPrice;
        }
    }
    public class Discount : Billing
    {
        public double GetDiscount()
        {
            double result = 0.0;
            if(ClothesType == "Women" && TotalPrice >=3000)
            {
                result = (TotalPrice * 12) / 100;
                result = TotalPrice - result;
            }
            else if (ClothesType == "Children" && TotalPrice >= 1500)
            {
                result = (TotalPrice * 7) / 100;
                result = TotalPrice - result;
            }
            else if (ClothesType == "Men" && TotalPrice >= 2000)
            {
                result = (TotalPrice * 10) / 100;
                result = TotalPrice - result;
            }
            else
            {
                return TotalPrice;
            }	 	  	    	 	      	  	 	
            return result;
        }
    }
